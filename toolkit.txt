ll bitcount(ll x)
{
	ll rv = 0;
	while(x > 0) { rv++; x &= (x - 1); }
	return rv;
}

void customsort()
{
    vector<ll> x;
    sort(x.begin(), x.end(),
        &[] (const ll &a, const ll &b) -> bool
        { return a < b; });
    // equivalent to regular sort
}

ll gcd(ll x, ll y)
{
	if(!y) { return x; }
	else { return gcd(y, x % y); }
}

ll pow(ll x, ll y)
{
    if(!y) return 1;
    ll rv = pow(x, y / 2);
    rv = (rv * rv) % MOD;
    if(y % 2) rv = (rv * x) % MOD;
    return rv;
}
